59 /**********************************************************************
60  * add_seam_to_queue
61  *
62  * Adds the given new_seam to the seams priority queue, unless it is full
63  * and the new seam is worse than the worst.
64  **********************************************************************/
65 void Wordrec::add_seam_to_queue(float new_priority, SEAM *new_seam,
66  SeamQueue* seams) {
67  if (new_seam == NULL) return;
68  if (chop_debug) {
69  tprintf("Pushing new seam with priority %g :", new_priority);
70  print_seam("seam: ", new_seam);
71  }
72  if (seams->size() >= MAX_NUM_SEAMS) {
73  SeamPair old_pair(0, NULL);
74  if (seams->PopWorst(&old_pair) && old_pair.key() <= new_priority) {
75  if (chop_debug) {
76  tprintf("Old seam staying with priority %g\n", old_pair.key());
77  }
78  delete new_seam;
79  seams->Push(&old_pair);
80  return;
81  } else if (chop_debug) {
82  tprintf("New seam with priority %g beats old worst seam with %g\n",
83  new_priority, old_pair.key());
84  }
85  }
86  SeamPair new_pair(new_priority, new_seam);
87  seams->Push(&new_pair);
88 }

91 /**********************************************************************
92  * choose_best_seam
93  *
94  * Choose the best seam that can be created by assembling this a
95  * collection of splits. A queue of all the possible seams is
96  * maintained. Each new split received is placed in that queue with
97  * its partial priority value. These values in the seam queue are
98  * evaluated and combined until a good enough seam is found. If no
99  * further good seams are being found then this function returns to the
100  * caller, who will send more splits. If this function is called with
101  * a split of NULL, then no further splits can be supplied by the
102  * caller.
103  **********************************************************************/
104 void Wordrec::choose_best_seam(SeamQueue* seam_queue,
105  SPLIT *split,
106  PRIORITY priority,
107  SEAM **seam_result,
108  TBLOB *blob,
109  SeamPile* seam_pile) {
110  SEAM *seam;
111  char str[80];
112  float my_priority;
113  /* Add seam of split */
114  my_priority = priority;
115  if (split != NULL) {
116  TPOINT split_point = split->point1->pos;
117  split_point += split->point2->pos;
118  split_point /= 2;
119  seam = new SEAM(my_priority, split_point, split, NULL, NULL);
120  if (chop_debug > 1)
121  print_seam ("Partial priority ", seam);
122  add_seam_to_queue(my_priority, seam, seam_queue);
123 
124  if (my_priority > chop_good_split)
125  return;
126  }
127 
128  TBOX bbox = blob->bounding_box();
129  /* Queue loop */
130  while (!seam_queue->empty()) {
131  SeamPair seam_pair;
132  seam_queue->Pop(&seam_pair);
133  seam = seam_pair.extract_data();
134  /* Set full priority */
135  my_priority = seam_priority(seam, bbox.left(), bbox.right());
136  if (chop_debug) {
137  sprintf (str, "Full my_priority %0.0f, ", my_priority);
138  print_seam(str, seam);
139  }
140 
141  if ((*seam_result == NULL || (*seam_result)->priority > my_priority) &&
142  my_priority < chop_ok_split) {
143  /* No crossing */
144  if (constrained_split(seam->split1, blob)) {
145  delete *seam_result;
146  *seam_result = new SEAM(*seam);
147  (*seam_result)->priority = my_priority;
148  } else {
149  delete seam;
150  seam = NULL;
151  my_priority = BAD_PRIORITY;
152  }
153  }
154 
155  if (my_priority < chop_good_split) {
156  if (seam)
157  delete seam;
158  return; /* Made good answer */
159  }
160 
161  if (seam) {
162  /* Combine with others */
163  if (seam_pile->size() < chop_seam_pile_size) {
164  combine_seam(*seam_pile, seam, seam_queue);
165  SeamDecPair pair(seam_pair.key(), seam);
166  seam_pile->Push(&pair);
167  } else if (chop_new_seam_pile &&
168  seam_pile->size() == chop_seam_pile_size &&
169  seam_pile->PeekTop().key() > seam_pair.key()) {
170  combine_seam(*seam_pile, seam, seam_queue);
171  SeamDecPair pair;
172  seam_pile->Pop(&pair); // pop the worst.
173  // Replace the seam in pair (deleting the old one) with
174  // the new seam and score, then push back into the heap.
175  pair.set_key(seam_pair.key());
176  pair.set_data(seam);
177  seam_pile->Push(&pair);
178  } else {
179  delete seam;
180  }
181  }
182 
183  my_priority = seam_queue->empty() ? NO_FULL_PRIORITY
184  : seam_queue->PeekTop().key();
185  if ((my_priority > chop_ok_split) ||
186  (my_priority > chop_good_split && split))
187  return;
188  }
189 }

192 /**********************************************************************
193  * combine_seam
194  *
195  * Find other seams to combine with this one. The new seams that result
196  * from this union should be added to the seam queue. The return value
197  * tells whether or not any additional seams were added to the queue.
198  **********************************************************************/
199 void Wordrec::combine_seam(const SeamPile& seam_pile,
200  const SEAM* seam, SeamQueue* seam_queue) {
201  register inT16 dist;
202  inT16 bottom1, top1;
203  inT16 bottom2, top2;
204 
205  SEAM *new_one;
206  const SEAM *this_one;
207 
208  bottom1 = seam->split1->point1->pos.y;
209  if (seam->split1->point2->pos.y >= bottom1)
210  top1 = seam->split1->point2->pos.y;
211  else {
212  top1 = bottom1;
213  bottom1 = seam->split1->point2->pos.y;
214  }
215  if (seam->split2 != NULL) {
216  bottom2 = seam->split2->point1->pos.y;
217  if (seam->split2->point2->pos.y >= bottom2)
218  top2 = seam->split2->point2->pos.y;
219  else {
220  top2 = bottom2;
221  bottom2 = seam->split2->point2->pos.y;
222  }
223  }
224  else {
225  bottom2 = bottom1;
226  top2 = top1;
227  }
228  for (int x = 0; x < seam_pile.size(); ++x) {
229  this_one = seam_pile.get(x).data();
230  dist = seam->location.x - this_one->location.x;
231  if (-SPLIT_CLOSENESS < dist &&
232  dist < SPLIT_CLOSENESS &&
233  seam->priority + this_one->priority < chop_ok_split) {
234  inT16 split1_point1_y = this_one->split1->point1->pos.y;
235  inT16 split1_point2_y = this_one->split1->point2->pos.y;
236  inT16 split2_point1_y = 0;
237  inT16 split2_point2_y = 0;
238  if (this_one->split2) {
239  split2_point1_y = this_one->split2->point1->pos.y;
240  split2_point2_y = this_one->split2->point2->pos.y;
241  }
242  if (
244  (
245  /* this_one->split1 always exists */
246  (
247  ((split1_point1_y >= top1 && split1_point2_y >= top1) ||
248  (split1_point1_y <= bottom1 && split1_point2_y <= bottom1))
249  &&
250  ((split1_point1_y >= top2 && split1_point2_y >= top2) ||
251  (split1_point1_y <= bottom2 && split1_point2_y <= bottom2))
252  )
253  )
254  &&
255  (
256  this_one->split2 == NULL ||
257  (
258  ((split2_point1_y >= top1 && split2_point2_y >= top1) ||
259  (split2_point1_y <= bottom1 && split2_point2_y <= bottom1))
260  &&
261  ((split2_point1_y >= top2 && split2_point2_y >= top2) ||
262  (split2_point1_y <= bottom2 && split2_point2_y <= bottom2))
263  )
264  )
265  ) {
266  new_one = join_two_seams (seam, this_one);
267  if (new_one != NULL) {
268  if (chop_debug > 1)
269  print_seam ("Combo priority ", new_one);
270  add_seam_to_queue(new_one->priority, new_one, seam_queue);
271  }
272  }
273  }
274  }

299 /**********************************************************************
300  * pick_good_seam
301  *
302  * Find and return a good seam that will split this blob into two pieces.
303  * Work from the outlines provided.
304  **********************************************************************/
305 SEAM *Wordrec::pick_good_seam(TBLOB *blob) {
306  SeamPile seam_pile(chop_seam_pile_size);
307  EDGEPT *points[MAX_NUM_POINTS];
308  EDGEPT_CLIST new_points;
309  SEAM *seam = NULL;
310  TESSLINE *outline;
311  inT16 num_points = 0;
312 
313 #ifndef GRAPHICS_DISABLED
314  if (chop_debug > 2)
315  wordrec_display_splits.set_value(true);
316 
317  draw_blob_edges(blob);
318 #endif
319 
320  PointHeap point_heap(MAX_NUM_POINTS);
321  for (outline = blob->outlines; outline; outline = outline->next)
322  prioritize_points(outline, &point_heap);
323 
324  while (!point_heap.empty() && num_points < MAX_NUM_POINTS) {
325  points[num_points++] = point_heap.PeekTop().data;
326  point_heap.Pop(NULL);
327  }
328 
329  /* Initialize queue */
330  SeamQueue seam_queue(MAX_NUM_SEAMS);
331 
332  try_point_pairs(points, num_points, &seam_queue, &seam_pile, &seam, blob);
333  try_vertical_splits(points, num_points, &new_points,
334  &seam_queue, &seam_pile, &seam, blob);
335 
336  if (seam == NULL) {
337  choose_best_seam(&seam_queue, NULL, BAD_PRIORITY, &seam, blob, &seam_pile);
338  }
339  else if (seam->priority > chop_good_split) {
340  choose_best_seam(&seam_queue, NULL, seam->priority,
341  &seam, blob, &seam_pile);
342  }
343 
344  EDGEPT_C_IT it(&new_points);
345  for (it.mark_cycle_pt(); !it.cycled_list(); it.forward()) {
346  EDGEPT *inserted_point = it.data();
347  if (!point_used_by_seam(seam, inserted_point)) {
348  for (outline = blob->outlines; outline; outline = outline->next) {
349  if (outline->loop == inserted_point) {
350  outline->loop = outline->loop->next;
351  }
352  }
353  remove_edgept(inserted_point);
354  }
355  }
356 
357  if (seam) {
358  if (seam->priority > chop_ok_split) {
359  delete seam;
360  seam = NULL;
361  }
362 #ifndef GRAPHICS_DISABLED
363  else if (wordrec_display_splits) {
364  if (seam->split1)
365  mark_split (seam->split1);
366  if (seam->split2)
367  mark_split (seam->split2);
368  if (seam->split3)
369  mark_split (seam->split3);
370  if (chop_debug > 2) {
371  update_edge_window();
372  edge_window_wait();
373  }
374  }
375 #endif
376  }
377 
378  if (chop_debug)
379  wordrec_display_splits.set_value(false);
380 
381  return (seam);
382 }
