98 EDGEPT *restore_outline(EDGEPT *start) {
99  EDGEPT *srcpt;
100  EDGEPT *real_start;
101 
102  if (start == NULL)
103  return NULL;
104  srcpt = start;
105  do {
106  if (srcpt->flags[1] == 2)
107  break;
108  srcpt = srcpt->next;
109  }
110  while (srcpt != start);

178  if (repair_unchopped_blobs)
179  preserve_outline_tree (blob->outlines);
180  TBLOB *other_blob = TBLOB::ShallowCopy(*blob); /* Make new blob */
181  // Insert it into the word.
182  word->blobs.insert(other_blob, blob_number + 1);
183 
184  SEAM *seam = NULL;
185  if (prioritize_division) {
186  TPOINT location;
187  if (divisible_blob(blob, italic_blob, &location)) {
188  seam = new SEAM(0.0f, location, NULL, NULL, NULL);
189  }
190  }
191  if (seam == NULL)
192  seam = pick_good_seam(blob);
193  if (chop_debug) {
194  if (seam != NULL)
195  print_seam("Good seam picked=", seam);
196  else
197  tprintf("\n** no seam picked *** \n");
198  }
199  if (seam) {
200  apply_seam(blob, other_blob, italic_blob, seam);
201  }
202 
203  seam = CheckSeam(chop_debug, blob_number, word, blob, other_blob,
204  seams, seam);
205  if (seam == NULL) {
206  if (repair_unchopped_blobs)
207  restore_outline_tree(blob->outlines);
208  if (word->latin_script) {

210  TPOINT location;

215  apply_seam(blob, other_blob, italic_blob, seam);
216  seam = CheckSeam(chop_debug, blob_number, word, blob, other_blob,
217  seams, seam);
218  }
219  }
220  }
221  return seam;

233 SEAM *Wordrec::chop_overlapping_blob(const GenericVector<TBOX>& boxes,
234  bool italic_blob, WERD_RES *word_res,
235  int *blob_number) {
236  TWERD *word = word_res->chopped_word;
237  for (*blob_number = 0; *blob_number < word->NumBlobs(); ++*blob_number) {
238  TBLOB *blob = word->blobs[*blob_number];
239  TPOINT topleft, botright;
240  topleft.x = blob->bounding_box().left();
241  topleft.y = blob->bounding_box().top();
242  botright.x = blob->bounding_box().right();
243  botright.y = blob->bounding_box().bottom();
244 
245  TPOINT original_topleft, original_botright;
246  word_res->denorm.DenormTransform(NULL, topleft, &original_topleft);
247  word_res->denorm.DenormTransform(NULL, botright, &original_botright);
248 
249  TBOX original_box = TBOX(original_topleft.x, original_botright.y,
250  original_botright.x, original_topleft.y);
251 
252  bool almost_equal_box = false;
253  int num_overlap = 0;
254  for (int i = 0; i < boxes.size(); i++) {
255  if (original_box.overlap_fraction(boxes[i]) > 0.125)
256  num_overlap++;
257  if (original_box.almost_equal(boxes[i], 3))
258  almost_equal_box = true;
259  }
260 
261  TPOINT location;
262  if (divisible_blob(blob, italic_blob, &location) ||
263  (!almost_equal_box && num_overlap > 1)) {
264  SEAM *seam = attempt_blob_chop(word, blob, *blob_number,
265  italic_blob, word_res->seam_array);
266  if (seam != NULL)
267  return seam;
268  }
269  }
270 
271  *blob_number = -1;
272  return NULL;
273 }

397 inT16 check_seam_order(TBLOB *blob, SEAM *seam) {
398  TESSLINE *outline;
399  inT8 found_em[3];
400 
401  if (seam->split1 == NULL || blob == NULL)
402  return (TRUE);
403 
404  found_em[0] = found_em[1] = found_em[2] = FALSE;
405 
406  for (outline = blob->outlines; outline; outline = outline->next) {
407  if (!found_em[0] &&
408  ((seam->split1 == NULL) ||
409  is_split_outline (outline, seam->split1))) {
410  found_em[0] = TRUE;
411  }
412  if (!found_em[1] &&
413  ((seam->split2 == NULL) ||
414  is_split_outline (outline, seam->split2))) {
415  found_em[1] = TRUE;
416  }
417  if (!found_em[2] &&
418  ((seam->split3 == NULL) ||
419  is_split_outline (outline, seam->split3))) {
420  found_em[2] = TRUE;
421  }
422  }
423 
424  if (!found_em[0] || !found_em[1] || !found_em[2])
425  return (FALSE);
426  else
427  return (TRUE);
428 }

440 void Wordrec::chop_word_main(WERD_RES *word) {
441  int num_blobs = word->chopped_word->NumBlobs();
442  if (word->ratings == NULL) {
443  word->ratings = new MATRIX(num_blobs, wordrec_max_join_chunks);
444  }
445  if (word->ratings->get(0, 0) == NULL) {
446  // Run initial classification.
447  for (int b = 0; b < num_blobs; ++b) {
448  BLOB_CHOICE_LIST* choices = classify_piece(word->seam_array, b, b,
449  "Initial:", word->chopped_word,
450  word->blamer_bundle);
451  word->ratings->put(b, b, choices);
452  }
453  } else {
454  // Blobs have been pre-classified. Set matrix cell for all blob choices
455  for (int col = 0; col < word->ratings->dimension(); ++col) {
456  for (int row = col; row < word->ratings->dimension() &&
457  row < col + word->ratings->bandwidth(); ++row) {
458  BLOB_CHOICE_LIST* choices = word->ratings->get(col, row);
459  if (choices != NULL) {
460  BLOB_CHOICE_IT bc_it(choices);
461  for (bc_it.mark_cycle_pt(); !bc_it.cycled_list(); bc_it.forward()) {
462  bc_it.data()->set_matrix_cell(col, row);
463  }
464  }
465  }
466  }
467  }
468 
469  // Run Segmentation Search.
470  BestChoiceBundle best_choice_bundle(word->ratings->dimension());
471  SegSearch(word, &best_choice_bundle, word->blamer_bundle);
472 
473  if (word->best_choice == NULL) {
474  // SegSearch found no valid paths, so just use the leading diagonal.
475  word->FakeWordFromRatings();
476  }
477  word->RebuildBestState();
478  // If we finished without a hyphen at the end of the word, let the next word
479  // be found in the dictionary.
480  if (word->word->flag(W_EOL) &&
481  !getDict().has_hyphen_end(*word->best_choice)) {
482  getDict().reset_hyphen_vars(true);
483  }
484 
485  if (word->blamer_bundle != NULL && this->fill_lattice_ != NULL) {
486  CallFillLattice(*word->ratings, word->best_choices,
487  *word->uch_set, word->blamer_bundle);
488  }
489  if (wordrec_debug_level > 0) {
490  tprintf("Final Ratings Matrix:\n");
491  word->ratings->print(getDict().getUnicharset());
492  }
493  word->FilterWordChoices(getDict().stopper_debug_level);
494 }

503 void Wordrec::improve_by_chopping(float rating_cert_scale,
504  WERD_RES* word,
505  BestChoiceBundle* best_choice_bundle,
506  BlamerBundle* blamer_bundle,
507  LMPainPoints* pain_points,
508  GenericVector<SegSearchPending>* pending) {
509  int blob_number;
510  do { // improvement loop.
511  // Make a simple vector of BLOB_CHOICEs to make it easy to pick which
512  // one to chop.
513  GenericVector<BLOB_CHOICE*> blob_choices;
514  int num_blobs = word->ratings->dimension();
515  for (int i = 0; i < num_blobs; ++i) {
516  BLOB_CHOICE_LIST* choices = word->ratings->get(i, i);
517  if (choices == NULL || choices->empty()) {
518  blob_choices.push_back(NULL);
519  } else {
520  BLOB_CHOICE_IT bc_it(choices);
521  blob_choices.push_back(bc_it.data());
522  }
523  }
524  SEAM* seam = improve_one_blob(blob_choices, &best_choice_bundle->fixpt,
525  false, false, word, &blob_number);
526  if (seam == NULL) break;
527  // A chop has been made. We have to correct all the data structures to
528  // take into account the extra bottom-level blob.
529  // Put the seam into the seam_array and correct everything else on the
530  // word: ratings matrix (including matrix location in the BLOB_CHOICES),
531  // states in WERD_CHOICEs, and blob widths.
532  word->InsertSeam(blob_number, seam);
533  // Insert a new entry in the beam array.
534  best_choice_bundle->beam.insert(new LanguageModelState, blob_number);
535  // Fixpts are outdated, but will get recalculated.
536  best_choice_bundle->fixpt.clear();
537  // Remap existing pain points.
538  pain_points->RemapForSplit(blob_number);
539  // Insert a new pending at the chop point.
540  pending->insert(SegSearchPending(), blob_number);
541 
542  // Classify the two newly created blobs using ProcessSegSearchPainPoint,
543  // as that updates the pending correctly and adds new pain points.
544  MATRIX_COORD pain_point(blob_number, blob_number);
545  ProcessSegSearchPainPoint(0.0f, pain_point, "Chop1", pending, word,
546  pain_points, blamer_bundle);
547  pain_point.col = blob_number + 1;
548  pain_point.row = blob_number + 1;
549  ProcessSegSearchPainPoint(0.0f, pain_point, "Chop2", pending, word,
550  pain_points, blamer_bundle);
551  if (language_model_->language_model_ngram_on) {
552  // N-gram evaluation depends on the number of blobs in a chunk, so we
553  // have to re-evaluate everything in the word.
554  ResetNGramSearch(word, best_choice_bundle, pending);
555  blob_number = 0;
556  }
557  // Run language model incrementally. (Except with the n-gram model on.)
558  UpdateSegSearchNodes(rating_cert_scale, blob_number, pending,
559  word, pain_points, best_choice_bundle, blamer_bundle);
560  } while (!language_model_->AcceptableChoiceFound() &&
561  word->ratings->dimension() < kMaxNumChunks);
562 
563  // If after running only the chopper best_choice is incorrect and no blame
564  // has been yet set, blame the classifier if best_choice is classifier's
565  // top choice and is a dictionary word (i.e. language model could not have
566  // helped). Otherwise blame the tradeoff between the classifier and
567  // the old language model (permuters).
568  if (word->blamer_bundle != NULL &&
569  word->blamer_bundle->incorrect_result_reason() == IRR_CORRECT &&
570  !word->blamer_bundle->ChoiceIsCorrect(word->best_choice)) {
571  bool valid_permuter = word->best_choice != NULL &&
572  Dict::valid_word_permuter(word->best_choice->permuter(), false);
573  word->blamer_bundle->BlameClassifierOrLangModel(word,
574  getDict().getUnicharset(),
575  valid_permuter,
576  wordrec_debug_blamer);
577  }
578 }

581 /**********************************************************************
582  * select_blob_to_split
583  *
584  * These are the results of the last classification. Find a likely
585  * place to apply splits. If none, return -1.
586  **********************************************************************/
587 int Wordrec::select_blob_to_split(
588  const GenericVector<BLOB_CHOICE*>& blob_choices,
589  float rating_ceiling, bool split_next_to_fragment) {
590  BLOB_CHOICE *blob_choice;
591  int x;
592  float worst = -MAX_FLOAT32;
593  int worst_index = -1;
594  float worst_near_fragment = -MAX_FLOAT32;
595  int worst_index_near_fragment = -1;
596  const CHAR_FRAGMENT **fragments = NULL;
597 
598  if (chop_debug) {
599  if (rating_ceiling < MAX_FLOAT32)
600  tprintf("rating_ceiling = %8.4f\n", rating_ceiling);
601  else
602  tprintf("rating_ceiling = No Limit\n");
603  }
604 
605  if (split_next_to_fragment && blob_choices.size() > 0) {
606  fragments = new const CHAR_FRAGMENT *[blob_choices.length()];
607  if (blob_choices[0] != NULL) {
608  fragments[0] = getDict().getUnicharset().get_fragment(
609  blob_choices[0]->unichar_id());
610  } else {
611  fragments[0] = NULL;
612  }
613  }
614 
615  for (x = 0; x < blob_choices.size(); ++x) {
616  if (blob_choices[x] == NULL) {
617  if (fragments != NULL) {
618  delete[] fragments;
619  }
620  return x;
621  } else {
622  blob_choice = blob_choices[x];
623  // Populate fragments for the following position.
624  if (split_next_to_fragment && x+1 < blob_choices.size()) {
625  if (blob_choices[x + 1] != NULL) {
626  fragments[x + 1] = getDict().getUnicharset().get_fragment(
627  blob_choices[x + 1]->unichar_id());
628  } else {
629  fragments[x + 1] = NULL;
630  }
631  }
632  if (blob_choice->rating() < rating_ceiling &&
633  blob_choice->certainty() < tessedit_certainty_threshold) {
634  // Update worst and worst_index.
635  if (blob_choice->rating() > worst) {
636  worst_index = x;
637  worst = blob_choice->rating();
638  }
639  if (split_next_to_fragment) {
640  // Update worst_near_fragment and worst_index_near_fragment.
641  bool expand_following_fragment =
642  (x + 1 < blob_choices.size() &&
643  fragments[x+1] != NULL && !fragments[x+1]->is_beginning());
644  bool expand_preceding_fragment =
645  (x > 0 && fragments[x-1] != NULL && !fragments[x-1]->is_ending());
646  if ((expand_following_fragment || expand_preceding_fragment) &&
647  blob_choice->rating() > worst_near_fragment) {
648  worst_index_near_fragment = x;
649  worst_near_fragment = blob_choice->rating();
650  if (chop_debug) {
651  tprintf("worst_index_near_fragment=%d"
652  " expand_following_fragment=%d"
653  " expand_preceding_fragment=%d\n",
654  worst_index_near_fragment,
655  expand_following_fragment,
656  expand_preceding_fragment);
657  }
658  }
659  }
660  }
661  }
662  }
663  if (fragments != NULL) {
664  delete[] fragments;
665  }
666  // TODO(daria): maybe a threshold of badness for
667  // worst_near_fragment would be useful.
668  return worst_index_near_fragment != -1 ?
669  worst_index_near_fragment : worst_index;
670 }

672 /**********************************************************************
673  * select_blob_to_split_from_fixpt
674  *
675  * Given the fix point from a dictionary search, if there is a single
676  * dangerous blob that maps to multiple characters, return that blob
677  * index as a place we need to split. If none, return -1.

679 int Wordrec::select_blob_to_split_from_fixpt(DANGERR *fixpt) {
680  if (!fixpt)
