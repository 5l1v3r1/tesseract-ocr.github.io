57

66  /* filter out duplicate points */

108  FLOAT32 MinSlope,
109  FLOAT32 MaxSlope) {
110 /*
111  ** Parameters:
112  ** Outline micro-feature outline to analyze

114  ** MaxSlope controls "snapping" of segments to vertical
115  ** Globals: none
116  ** Operation:
117  ** This routine searches thru the specified outline, computes
118  ** a slope for each vector in the outline, and marks each
119  ** vector as having one of the following directions:

403 void ComputeDirection(MFEDGEPT *Start,
404  MFEDGEPT *Finish,
405  FLOAT32 MinSlope,
406  FLOAT32 MaxSlope) {
407 /*
408  ** Parameters:
409  ** Start starting point to compute direction from
410  ** Finish finishing point to compute direction to
411  ** MinSlope slope below which lines are horizontal
412  ** MaxSlope slope above which lines are vertical
413  ** Globals: none
414  ** Operation:
415  ** This routine computes the slope from Start to Finish and
416  ** and then computes the approximate direction of the line
417  ** segment from Start to Finish. The direction is quantized
418  ** into 8 buckets:
419  ** N, S, E, W, NE, NW, SE, SW
420  ** Both the slope and the direction are then stored into
421  ** the appropriate fields of the Start edge point. The
422  ** direction is also stored into the PreviousDirection field
423  ** of the Finish edge point.
424  ** Return: none
425  ** Exceptions: none
426  ** History: 7/25/89, DSJ, Created.
427  */
428  FVECTOR Delta;
429 
430  Delta.x = Finish->Point.x - Start->Point.x;
431  Delta.y = Finish->Point.y - Start->Point.y;
432  if (Delta.x == 0)
433  if (Delta.y < 0) {
434  Start->Slope = -MAX_FLOAT32;
435  Start->Direction = south;
436  }
437  else {
438  Start->Slope = MAX_FLOAT32;
439  Start->Direction = north;
440  }
441  else {
442  Start->Slope = Delta.y / Delta.x;
443  if (Delta.x > 0)
444  if (Delta.y > 0)
445  if (Start->Slope > MinSlope)
446  if (Start->Slope < MaxSlope)
447  Start->Direction = northeast;
448  else
449  Start->Direction = north;
450  else
451  Start->Direction = east;
452  else if (Start->Slope < -MinSlope)
453  if (Start->Slope > -MaxSlope)
454  Start->Direction = southeast;
455  else
456  Start->Direction = south;
457  else
458  Start->Direction = east;
459  else if (Delta.y > 0)
460  if (Start->Slope < -MinSlope)
461  if (Start->Slope > -MaxSlope)
462  Start->Direction = northwest;
463  else
464  Start->Direction = north;
465  else
466  Start->Direction = west;
467  else if (Start->Slope > MinSlope)
468  if (Start->Slope < MaxSlope)
469  Start->Direction = southwest;
470  else
471  Start->Direction = south;
472  else
473  Start->Direction = west;
474  }
475  Finish->PreviousDirection = Start->Direction;
476 } /* ComputeDirection */

480 void FinishOutlineStats(register OUTLINE_STATS *OutlineStats) {
481 /*
482  ** Parameters:
483  ** OutlineStats statistics about a set of outlines
484  ** Globals: none
485  ** Operation: Use the preliminary statistics accumulated in OutlineStats
486  ** to compute the final statistics.
487  ** (see Dan Johnson's Tesseract lab
488  ** notebook #2, pgs. 74-78).
489  ** Return: none
490  ** Exceptions: none
491  ** History: Fri Dec 14 10:13:36 1990, DSJ, Created.
492  */
493  OutlineStats->x = 0.5 * OutlineStats->My / OutlineStats->L;
494  OutlineStats->y = 0.5 * OutlineStats->Mx / OutlineStats->L;
495 
496  OutlineStats->Ix = (OutlineStats->Ix / 3.0 -
497  OutlineStats->y * OutlineStats->Mx +
498  OutlineStats->y * OutlineStats->y * OutlineStats->L);
499 
500  OutlineStats->Iy = (OutlineStats->Iy / 3.0 -
501  OutlineStats->x * OutlineStats->My +
502  OutlineStats->x * OutlineStats->x * OutlineStats->L);
503 
504  /* Ix and/or Iy could possibly be negative due to roundoff error */
505  if (OutlineStats->Ix < 0.0)
506  OutlineStats->Ix = MIN_INERTIA;
507  if (OutlineStats->Iy < 0.0)
508  OutlineStats->Iy = MIN_INERTIA;
509 
510  OutlineStats->Rx = sqrt (OutlineStats->Ix / OutlineStats->L);
511  OutlineStats->Ry = sqrt (OutlineStats->Iy / OutlineStats->L);
512 
513  OutlineStats->Mx *= 0.5;
514  OutlineStats->My *= 0.5;
515 
516 } /* FinishOutlineStats */

575 void UpdateOutlineStats(register OUTLINE_STATS *OutlineStats,
576  register FLOAT32 x1,
577  register FLOAT32 x2,
578  register FLOAT32 y1,
579  register FLOAT32 y2) {
580 /*
581  ** Parameters:
582  ** OutlineStats statistics to add this segment to
583  ** x1, y1, x2, y2 segment to be added to statistics
584  ** Globals: none
585  ** Operation: This routine adds the statistics for the specified
586  ** line segment to OutlineStats. The statistics that are
587  ** kept are:
588  ** sum of length of all segments
589  ** sum of 2*Mx for all segments
590  ** sum of 2*My for all segments
591  ** sum of 2*Mx*(y1+y2) - L*y1*y2 for all segments
592  ** sum of 2*My*(x1+x2) - L*x1*x2 for all segments
593  ** These numbers, once collected can later be used to easily
594  ** compute the center of mass, first and second moments,
595  ** and radii of gyration. (see Dan Johnson's Tesseract lab
596  ** notebook #2, pgs. 74-78).
597  ** Return: none
598  ** Exceptions: none
599  ** History: Fri Dec 14 08:59:17 1990, DSJ, Created.
600  */
601  register FLOAT64 L;
602  register FLOAT64 Mx2;
603  register FLOAT64 My2;
604 
605  /* compute length of segment */
606  L = sqrt ((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
607  OutlineStats->L += L;
608 
609  /* compute 2Mx and 2My components */
610  Mx2 = L * (y1 + y2);
611  My2 = L * (x1 + x2);
612  OutlineStats->Mx += Mx2;
613  OutlineStats->My += My2;
614 
615  /* compute second moment component */
616  OutlineStats->Ix += Mx2 * (y1 + y2) - L * y1 * y2;
617  OutlineStats->Iy += My2 * (x1 + x2) - L * x1 * x2;
618 
619 } /* UpdateOutlineStats */
