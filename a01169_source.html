93 void BoxChar::WriteTesseractBoxFile(const string& filename, int height,
94  const vector<BoxChar*>& boxes) {
95  string output;
96  char buffer[kMaxLineLength];
97  for (int i = 0; i < boxes.size(); ++i) {
98  const Box* box = boxes[i]->box_;
99  if (box != NULL) {
100  if (i > 0 && boxes[i - 1]->box_ != NULL &&
101  boxes[i - 1]->page_ == boxes[i]->page_ &&
102  box->x + box->w < boxes[i - 1]->box_->x) {
103  // We are on a newline. Output a tab character to indicate the newline.
104  AppendTabBox(boxes[i - 1]->box_, height, boxes[i]->page_, &output);
105  }
106  int nbytes = snprintf(buffer, kMaxLineLength,
107  "%s %d %d %d %d %d\n",
108  boxes[i]->ch_.c_str(),
109  box->x, height - box->y - box->h,
110  box->x + box->w, height - box->y,
111  boxes[i]->page_);
112  output.append(buffer, nbytes);
113  } else if (i > 0 && boxes[i - 1]->box_ != NULL) {
114  int j = i + 1;
115  // Find the next non-null box, as there may be multiple spaces.
116  while (j < boxes.size() && boxes[j]->box_ == NULL) ++j;
117  if (j < boxes.size() && boxes[i - 1]->page_ == boxes[j]->page_) {
118  const Box* prev = boxes[i - 1]->box_;
119  const Box* next = boxes[j]->box_;
120  if (next->x + next->w < prev->x) {
121  // We are on a newline. Output a tab character to indicate it.
122  AppendTabBox(prev, height, boxes[j]->page_, &output);
123  } else {
124  // Space between words.
125  int nbytes = snprintf(buffer, kMaxLineLength,
126  " %d %d %d %d %d\n",
127  prev->x + prev->w,
128  height - MAX(prev->y + prev->h,
129  next->y + next->h),
130  next->x, height - MIN(prev->y, next->y),
131  boxes[i - 1]->page_);
132  output.append(buffer, nbytes);
133  }
134  }
135  }
136  }
137  File::WriteStringToFileOrDie(output, filename);
138 }
