67 // Consume the circular list of EDGEPTs to make a TESSLINE.
68 TESSLINE* TESSLINE::BuildFromOutlineList(EDGEPT* outline) {
69  TESSLINE* result = new TESSLINE;
70  result->loop = outline;
71  if (outline->src_outline != NULL) {
72  // ASSUMPTION: This function is only ever called from ApproximateOutline
73  // and therefore either all points have a src_outline or all do not.
74  // Just as SetupFromPos sets the vectors from the vertices, setup the
75  // step_count members to indicate the (positive) number of original
76  // C_OUTLINE steps to the next vertex.
77  EDGEPT* pt = outline;
78  do {
79  pt->step_count = pt->next->start_step - pt->start_step;
80  if (pt->step_count < 0)
81  pt->step_count += pt->src_outline->pathlength();
82  pt = pt->next;
83  } while (pt != outline);
84  }
85  result->SetupFromPos();
86  return result;
87 }

188 // Recomputes the bounding box from the points in the loop.
189 void TESSLINE::ComputeBoundingBox() {
190  int minx = MAX_INT32;
191  int miny = MAX_INT32;
192  int maxx = -MAX_INT32;
193  int maxy = -MAX_INT32;
194 
195  // Find boundaries.
196  start = loop->pos;
197  EDGEPT* this_edge = loop;
198  do {
199  if (!this_edge->IsHidden() || !this_edge->prev->IsHidden()) {
200  if (this_edge->pos.x < minx)
201  minx = this_edge->pos.x;
202  if (this_edge->pos.y < miny)
203  miny = this_edge->pos.y;
204  if (this_edge->pos.x > maxx)
205  maxx = this_edge->pos.x;
206  if (this_edge->pos.y > maxy)
207  maxy = this_edge->pos.y;
208  }
209  this_edge = this_edge->next;
210  } while (this_edge != loop);
211  // Reset bounds.
212  topleft.x = minx;
213  topleft.y = maxy;
214  botright.x = maxx;
215  botright.y = miny;
216 }

218 // Computes the min and max cross product of the outline points with the
219 // given vec and returns the results in min_xp and max_xp. Geometrically
220 // this is the left and right edge of the outline perpendicular to the
221 // given direction, but to get the distance units correct, you would
222 // have to divide by the modulus of vec.
223 void TESSLINE::MinMaxCrossProduct(const TPOINT vec,
224  int* min_xp, int* max_xp) const {
225  *min_xp = MAX_INT32;
226  *max_xp = MIN_INT32;
227  EDGEPT* this_edge = loop;
228  do {
229  if (!this_edge->IsHidden() || !this_edge->prev->IsHidden()) {
230  int product = CROSS(this_edge->pos, vec);
231  UpdateRange(product, min_xp, max_xp);
232  }
233  this_edge = this_edge->next;
234  } while (this_edge != loop);
235 }

280 // Iterate the given list of outlines, converting to TESSLINE by polygonal
281 // approximation and recursively any children, returning the current tail
282 // of the resulting list of TESSLINEs.
283 static TESSLINE** ApproximateOutlineList(bool allow_detailed_fx,
284  C_OUTLINE_LIST* outlines,
285  bool children,
286  TESSLINE** tail) {
287  C_OUTLINE_IT ol_it(outlines);
288  for (ol_it.mark_cycle_pt(); !ol_it.cycled_list(); ol_it.forward()) {
289  C_OUTLINE* outline = ol_it.data();
290  if (outline->pathlength() > 0) {
291  TESSLINE* tessline = ApproximateOutline(allow_detailed_fx, outline);
292  tessline->is_hole = children;
293  *tail = tessline;
294  tail = &tessline->next;
295  }
296  if (!outline->child()->empty()) {
297  tail = ApproximateOutlineList(allow_detailed_fx, outline->child(), true,
298  tail);
299  }
300  }
301  return tail;
302 }

327 TBLOB* TBLOB::ClassifyNormalizeIfNeeded() const {

330  // +/- 90 degrees, as 180 was already taken care of.
331  if (denorm_.block() != NULL &&

465 // Computes the center of mass and second moments for the old baseline and
466 // 2nd moment normalizations. Returns the outline length.
467 // The input denorm should be the normalizations that have been applied from
468 // the image to the current state of this TBLOB.
469 int TBLOB::ComputeMoments(FCOORD* center, FCOORD* second_moments) const {
470  // Compute 1st and 2nd moments of the original outline.
471  LLSQ accumulator;
472  TBOX box = bounding_box();
473  // Iterate the outlines, accumulating edges relative the box.botleft().
474  CollectEdges(box, NULL, &accumulator, NULL, NULL);
475  *center = accumulator.mean_point() + box.botleft();
476  // The 2nd moments are just the standard deviation of the point positions.
477  double x2nd = sqrt(accumulator.x_variance());
478  double y2nd = sqrt(accumulator.y_variance());
479  if (x2nd < 1.0) x2nd = 1.0;
480  if (y2nd < 1.0) y2nd = 1.0;
481  second_moments->set_x(x2nd);
482  second_moments->set_y(y2nd);
483  return accumulator.count();
484 }

518 // Accumulates the segment between pt1 and pt2 in the LLSQ, quantizing over
519 // the integer coordinate grid to properly weight long vectors.
520 static void SegmentLLSQ(const FCOORD& pt1, const FCOORD& pt2,
521  LLSQ* accumulator) {
522  FCOORD step(pt2);
523  step -= pt1;
524  int xstart = IntCastRounded(MIN(pt1.x(), pt2.x()));
525  int xend = IntCastRounded(MAX(pt1.x(), pt2.x()));
526  int ystart = IntCastRounded(MIN(pt1.y(), pt2.y()));
527  int yend = IntCastRounded(MAX(pt1.y(), pt2.y()));
528  if (xstart == xend && ystart == yend) return; // Nothing to do.
529  double weight = step.length() / (xend - xstart + yend - ystart);
530  // Compute and save the y-position at the middle of each x-step.
531  for (int x = xstart; x < xend; ++x) {
532  double y = pt1.y() + step.y() * (x + 0.5 - pt1.x()) / step.x();
533  accumulator->add(x + 0.5, y, weight);
534  }
535  // Compute and save the x-position at the middle of each y-step.
536  for (int y = ystart; y < yend; ++y) {
537  double x = pt1.x() + step.x() * (y + 0.5 - pt1.y()) / step.y();
538  accumulator->add(x, y + 0.5, weight);
539  }
540 }

542 // Adds any edges from a single segment of outline between pt1 and pt2 to
543 // the x_coords, y_coords vectors. pt1 and pt2 should be relative to the
544 // bottom-left of the bounding box, hence indices to x_coords, y_coords
545 // are clipped to ([0,x_limit], [0,y_limit]).
546 // See GetEdgeCoords above for a description of x_coords, y_coords.
547 static void SegmentCoords(const FCOORD& pt1, const FCOORD& pt2,
548  int x_limit, int y_limit,
549  GenericVector<GenericVector<int> >* x_coords,
550  GenericVector<GenericVector<int> >* y_coords) {
551  FCOORD step(pt2);
552  step -= pt1;
553  int start = ClipToRange(IntCastRounded(MIN(pt1.x(), pt2.x())), 0, x_limit);
554  int end = ClipToRange(IntCastRounded(MAX(pt1.x(), pt2.x())), 0, x_limit);
555  for (int x = start; x < end; ++x) {
556  int y = IntCastRounded(pt1.y() + step.y() * (x + 0.5 - pt1.x()) / step.x());
557  (*y_coords)[x].push_back(y);
558  }
559  start = ClipToRange(IntCastRounded(MIN(pt1.y(), pt2.y())), 0, y_limit);
560  end = ClipToRange(IntCastRounded(MAX(pt1.y(), pt2.y())), 0, y_limit);
561  for (int y = start; y < end; ++y) {
562  int x = IntCastRounded(pt1.x() + step.x() * (y + 0.5 - pt1.y()) / step.y());
563  (*x_coords)[y].push_back(x);
564  }
565 }

567 // Adds any edges from a single segment of outline between pt1 and pt2 to
568 // the bbox such that it guarantees to contain anything produced by
569 // SegmentCoords.
570 static void SegmentBBox(const FCOORD& pt1, const FCOORD& pt2, TBOX* bbox) {
571  FCOORD step(pt2);
572  step -= pt1;
573  int x1 = IntCastRounded(MIN(pt1.x(), pt2.x()));
574  int x2 = IntCastRounded(MAX(pt1.x(), pt2.x()));
575  if (x2 > x1) {
576  int y1 = IntCastRounded(pt1.y() + step.y() * (x1 + 0.5 - pt1.x()) /
577  step.x());
578  int y2 = IntCastRounded(pt1.y() + step.y() * (x2 - 0.5 - pt1.x()) /
579  step.x());
580  TBOX point(x1, MIN(y1, y2), x2, MAX(y1, y2));
581  *bbox += point;
582  }

585  if (y2 > y1) {

595 // Collects edges into the given bounding box, LLSQ accumulator and/or x_coords,
596 // y_coords vectors.
597 // For a description of x_coords/y_coords, see GetEdgeCoords above.
598 // Startpt to lastpt, inclusive, MUST have the same src_outline member,
599 // which may be NULL. The vector from lastpt to its next is included in
600 // the accumulation. Hidden edges should be excluded by the caller.
601 // The input denorm should be the normalizations that have been applied from
602 // the image to the current state of the TBLOB from which startpt, lastpt come.
603 // box is the bounding box of the blob from which the EDGEPTs are taken and
604 // indices into x_coords, y_coords are offset by box.botleft().
605 static void CollectEdgesOfRun(const EDGEPT* startpt, const EDGEPT* lastpt,
606  const DENORM& denorm, const TBOX& box,
607  TBOX* bounding_box,
608  LLSQ* accumulator,
609  GenericVector<GenericVector<int> > *x_coords,
610  GenericVector<GenericVector<int> > *y_coords) {
611  const C_OUTLINE* outline = startpt->src_outline;
612  int x_limit = box.width() - 1;
613  int y_limit = box.height() - 1;
614  if (outline != NULL) {
615  // Use higher-resolution edge points stored on the outline.
616  // The outline coordinates may not match the binary image because of the
617  // rotation for vertical text lines, but the root_denorm IS the matching
618  // start of the DENORM chain.
619  const DENORM* root_denorm = denorm.RootDenorm();
620  int step_length = outline->pathlength();
621  int start_index = startpt->start_step;
622  // Note that if this run straddles the wrap-around point of the outline,
623  // that lastpt->start_step may have a lower index than startpt->start_step,
624  // and we want to use an end_index that allows us to use a positive
625  // increment, so we add step_length if necessary, but that may be beyond the
626  // bounds of the outline steps/ due to wrap-around, so we use % step_length
627  // everywhere, except for start_index.
628  int end_index = lastpt->start_step + lastpt->step_count;
629  if (end_index <= start_index)
630  end_index += step_length;
631  // pos is the integer coordinates of the binary image steps.
632  ICOORD pos = outline->position_at_index(start_index);
633  FCOORD origin(box.left(), box.bottom());
634  // f_pos is a floating-point version of pos that offers improved edge
635  // positioning using greyscale information or smoothing of edge steps.
636  FCOORD f_pos = outline->sub_pixel_pos_at_index(pos, start_index);
637  // pos_normed is f_pos after the appropriate normalization, and relative
638  // to origin.
639  // prev_normed is the previous value of pos_normed.
640  FCOORD prev_normed;
641  denorm.NormTransform(root_denorm, f_pos, &prev_normed);
642  prev_normed -= origin;
643  for (int index = start_index; index < end_index; ++index) {
644  ICOORD step = outline->step(index % step_length);
645  // Only use the point if its edge strength is positive. This excludes
646  // points that don't provide useful information, eg
647  // ___________
648  // |___________
649  // The vertical step provides only noisy, damaging information, as even
650  // with a greyscale image, the positioning of the edge there may be a
651  // fictitious extrapolation, so previous processing has eliminated it.
652  if (outline->edge_strength_at_index(index % step_length) > 0) {
653  FCOORD f_pos = outline->sub_pixel_pos_at_index(pos,
654  index % step_length);
655  FCOORD pos_normed;
656  denorm.NormTransform(root_denorm, f_pos, &pos_normed);
657  pos_normed -= origin;
658  // Accumulate the information that is selected by the caller.
659  if (bounding_box != NULL) {
660  SegmentBBox(pos_normed, prev_normed, bounding_box);
661  }
662  if (accumulator != NULL) {
663  SegmentLLSQ(pos_normed, prev_normed, accumulator);
664  }
665  if (x_coords != NULL && y_coords != NULL) {
666  SegmentCoords(pos_normed, prev_normed, x_limit, y_limit,
667  x_coords, y_coords);
668  }
669  prev_normed = pos_normed;
670  }
671  pos += step;
672  }
673  } else {
674  // There is no outline, so we are forced to use the polygonal approximation.
675  const EDGEPT* endpt = lastpt->next;
676  const EDGEPT* pt = startpt;
677  do {
678  FCOORD next_pos(pt->next->pos.x - box.left(),
679  pt->next->pos.y - box.bottom());
680  FCOORD pos(pt->pos.x - box.left(), pt->pos.y - box.bottom());
681  if (bounding_box != NULL) {
682  SegmentBBox(next_pos, pos, bounding_box);
683  }
684  if (accumulator != NULL) {
685  SegmentLLSQ(next_pos, pos, accumulator);
686  }
687  if (x_coords != NULL && y_coords != NULL) {
688  SegmentCoords(next_pos, pos, x_limit, y_limit, x_coords, y_coords);
689  }
690  } while ((pt = pt->next) != endpt);
691  }
692 }
693 
694 // For all the edge steps in all the outlines, or polygonal approximation
695 // where there are no edge steps, collects the steps into the bounding_box,
696 // llsq and/or the x_coords/y_coords. Both are used in different kinds of
697 // normalization.
698 // For a description of x_coords, y_coords, see GetEdgeCoords above.
699 void TBLOB::CollectEdges(const TBOX& box,
700  TBOX* bounding_box, LLSQ* llsq,
701  GenericVector<GenericVector<int> >* x_coords,
702  GenericVector<GenericVector<int> >* y_coords) const {
703  // Iterate the outlines.
704  for (const TESSLINE* ol = outlines; ol != NULL; ol = ol->next) {
705  // Iterate the polygon.
706  EDGEPT* loop_pt = ol->FindBestStartPt();
707  EDGEPT* pt = loop_pt;
708  if (pt == NULL) continue;
709  do {
710  if (pt->IsHidden()) continue;
711  // Find a run of equal src_outline.
712  EDGEPT* last_pt = pt;
713  do {
714  last_pt = last_pt->next;
715  } while (last_pt != loop_pt && !last_pt->IsHidden() &&
716  last_pt->src_outline == pt->src_outline);
717  last_pt = last_pt->prev;
718  CollectEdgesOfRun(pt, last_pt, denorm_, box,
719  bounding_box, llsq, x_coords, y_coords);
720  pt = last_pt;
721  } while ((pt = pt->next) != loop_pt);
722  }
723 }

739 // Baseline normalizes the blobs in-place, recording the normalization in the
740 // DENORMs in the blobs.
741 void TWERD::BLNormalize(const BLOCK* block, const ROW* row, Pix* pix,
742  bool inverse, float x_height, bool numeric_mode,
743  tesseract::OcrEngineMode hint,
744  const TBOX* norm_box,
745  DENORM* word_denorm) {
746  TBOX word_box = bounding_box();
747  if (norm_box != NULL) word_box = *norm_box;
748  float word_middle = (word_box.left() + word_box.right()) / 2.0f;
749  float input_y_offset = 0.0f;
750  float final_y_offset = static_cast<float>(kBlnBaselineOffset);
751  float scale = kBlnXHeight / x_height;
752  if (hint == tesseract::OEM_CUBE_ONLY || row == NULL) {
753  word_middle = word_box.left();
754  input_y_offset = word_box.bottom();
755  final_y_offset = 0.0f;
756  if (hint == tesseract::OEM_CUBE_ONLY)
757  scale = 1.0f;
758  } else {
759  input_y_offset = row->base_line(word_middle);
760  }
761  for (int b = 0; b < blobs.size(); ++b) {
762  TBLOB* blob = blobs[b];
763  TBOX blob_box = blob->bounding_box();
764  float mid_x = (blob_box.left() + blob_box.right()) / 2.0f;
765  float baseline = input_y_offset;
766  float blob_scale = scale;
767  if (numeric_mode) {
768  baseline = blob_box.bottom();
769  blob_scale = ClipToRange(kBlnXHeight * 4.0f / (3 * blob_box.height()),
770  scale, scale * 1.5f);
771  } else if (row != NULL && hint != tesseract::OEM_CUBE_ONLY) {
772  baseline = row->base_line(mid_x);
773  }
774  // The image will be 8-bit grey if the input was grey or color. Note that in
775  // a grey image 0 is black and 255 is white. If the input was binary, then
776  // the pix will be binary and 0 is white, with 1 being black.
777  // To tell the difference pixGetDepth() will return 8 or 1.
778  // The inverse flag will be true iff the word has been determined to be
779  // white on black, and is independent of whether the pix is 8 bit or 1 bit.
780  blob->Normalize(block, NULL, NULL, word_middle, baseline, blob_scale,
781  blob_scale, 0.0f, final_y_offset, inverse, pix);
782  }
783  if (word_denorm != NULL) {
784  word_denorm->SetupNormalization(block, NULL, NULL, word_middle,
785  input_y_offset, scale, scale,
786  0.0f, final_y_offset);
787  word_denorm->set_inverse(inverse);
788  word_denorm->set_pix(pix);
789  }
790 }

824 // Merges the blobs from start to end, not including end, and deletes
825 // the blobs between start and end.
826 void TWERD::MergeBlobs(int start, int end) {
827  if (start >= blobs.size() - 1) return; // Nothing to do.
828  TESSLINE* outline = blobs[start]->outlines;
829  for (int i = start + 1; i < end && i < blobs.size(); ++i) {
830  TBLOB* next_blob = blobs[i];
831  // Take the outlines from the next blob.
832  if (outline == NULL) {
833  blobs[start]->outlines = next_blob->outlines;
834  outline = blobs[start]->outlines;
835  } else {
836  while (outline->next != NULL)
837  outline = outline->next;
838  outline->next = next_blob->outlines;
839  next_blob->outlines = NULL;
840  }
841  // Delete the next blob and move on.
842  delete next_blob;
843  blobs[i] = NULL;
844  }
845  // Remove dead blobs from the vector.
846  for (int i = start + 1; i < end && start + 1 < blobs.size(); ++i) {
847  blobs.remove(start + 1);
848  }

873 /**********************************************************************
874  * divisible_blob
875  *
876  * Returns true if the blob contains multiple outlines than can be
877  * separated using divide_blobs. Sets the location to be used in the
878  * call to divide_blobs.
879  **********************************************************************/
880 bool divisible_blob(TBLOB *blob, bool italic_blob, TPOINT* location) {
881  if (blob->outlines == NULL || blob->outlines->next == NULL)
882  return false; // Need at least 2 outlines for it to be possible.
883  int max_gap = 0;
884  TPOINT vertical = italic_blob ? kDivisibleVerticalItalic
885  : kDivisibleVerticalUpright;
886  for (TESSLINE* outline1 = blob->outlines; outline1 != NULL;
887  outline1 = outline1->next) {
888  if (outline1->is_hole)
889  continue; // Holes do not count as separable.
890  TPOINT mid_pt1(
891  static_cast<inT16>((outline1->topleft.x + outline1->botright.x) / 2),
892  static_cast<inT16>((outline1->topleft.y + outline1->botright.y) / 2));
893  int mid_prod1 = CROSS(mid_pt1, vertical);
894  int min_prod1, max_prod1;
895  outline1->MinMaxCrossProduct(vertical, &min_prod1, &max_prod1);
896  for (TESSLINE* outline2 = outline1->next; outline2 != NULL;
897  outline2 = outline2->next) {
898  if (outline2->is_hole)
899  continue; // Holes do not count as separable.
900  TPOINT mid_pt2(
901  static_cast<inT16>((outline2->topleft.x + outline2->botright.x) / 2),
902  static_cast<inT16>((outline2->topleft.y + outline2->botright.y) / 2));
903  int mid_prod2 = CROSS(mid_pt2, vertical);
904  int min_prod2, max_prod2;
905  outline2->MinMaxCrossProduct(vertical, &min_prod2, &max_prod2);
906  int mid_gap = abs(mid_prod2 - mid_prod1);
907  int overlap = MIN(max_prod1, max_prod2) - MAX(min_prod1, min_prod2);
908  if (mid_gap - overlap / 4 > max_gap) {
909  max_gap = mid_gap - overlap / 4;
910  *location = mid_pt1;
911  *location += mid_pt2;
912  *location /= 2;
913  }
914  }
915  }
916  // Use the y component of the vertical vector as an approximation to its
917  // length.
918  return max_gap > vertical.y;
919 }

921 /**********************************************************************
922  * divide_blobs
923  *
924  * Create two blobs by grouping the outlines in the appropriate blob.
925  * The outlines that are beyond the location point are moved to the
926  * other blob. The ones whose x location is less than that point are
927  * retained in the original blob.
928  **********************************************************************/
929 void divide_blobs(TBLOB *blob, TBLOB *other_blob, bool italic_blob,
930  const TPOINT& location) {
931  TPOINT vertical = italic_blob ? kDivisibleVerticalItalic
932  : kDivisibleVerticalUpright;
933  TESSLINE *outline1 = NULL;
934  TESSLINE *outline2 = NULL;
935 
936  TESSLINE *outline = blob->outlines;
937  blob->outlines = NULL;
938  int location_prod = CROSS(location, vertical);
939 
940  while (outline != NULL) {
941  TPOINT mid_pt(
942  static_cast<inT16>((outline->topleft.x + outline->botright.x) / 2),
943  static_cast<inT16>((outline->topleft.y + outline->botright.y) / 2));
944  int mid_prod = CROSS(mid_pt, vertical);
945  if (mid_prod < location_prod) {
946  // Outline is in left blob.
947  if (outline1)
948  outline1->next = outline;
949  else
950  blob->outlines = outline;
951  outline1 = outline;
952  } else {
953  // Outline is in right blob.
954  if (outline2)
955  outline2->next = outline;
956  else
957  other_blob->outlines = outline;
958  outline2 = outline;
959  }
960  outline = outline->next;
961  }
962 
963  if (outline1)
964  outline1->next = NULL;
965  if (outline2)
966  outline2->next = NULL;
967 }
