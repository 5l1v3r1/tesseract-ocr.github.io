75  // Fills spacing with the value of the x gap expected between the two given
76  // UNICHAR_IDs. Returns true on success.
77  bool get_spacing(UNICHAR_ID prev_uch_id,
78  UNICHAR_ID uch_id,
79  int *spacing) const {
80  const FontSpacingInfo *prev_fsi = this->get_spacing(prev_uch_id);
81  const FontSpacingInfo *fsi = this->get_spacing(uch_id);
82  if (prev_fsi == NULL || fsi == NULL) return false;
83  int i = 0;
84  for (; i < prev_fsi->kerned_unichar_ids.size(); ++i) {
85  if (prev_fsi->kerned_unichar_ids[i] == uch_id) break;
86  }
87  if (i < prev_fsi->kerned_unichar_ids.size()) {
88  *spacing = prev_fsi->kerned_x_gaps[i];
89  } else {
90  *spacing = prev_fsi->x_gap_after + fsi->x_gap_before;
91  }
92  return true;
93  }

112 // Every class (character) owns a FontSet that represents all the fonts that can
113 // render this character.
114 // Since almost all the characters from the same script share the same set of
115 // fonts, the sets are shared over multiple classes (see
116 // Classify::fontset_table_). Thus, a class only store an id to a set.
117 // Because some fonts cannot render just one character of a set, there are a
118 // lot of FontSet that differ only by one font. Rather than storing directly
119 // the FontInfo in the FontSet structure, it's better to share FontInfos among
120 // FontSets (Classify::fontinfo_table_).
121 struct FontSet {
122  int size;
123  int* configs; // FontInfo ids
124 };
