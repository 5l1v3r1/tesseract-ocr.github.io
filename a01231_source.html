95 PRIORITY Wordrec::grade_center_of_blob(register BOUNDS_RECT rect) {

115 /**********************************************************************
116  * grade_overlap
117  *
118  * Return a grade for this split for the overlap of the resultant blobs.

122 PRIORITY Wordrec::grade_overlap(register BOUNDS_RECT rect) {
123  register PRIORITY grade;
124  register inT16 width1;
125  register inT16 width2;
126  register inT16 overlap;
127 
128  width1 = rect[3] - rect[2];

132  width1 = MIN (width1, width2);
