191  // Inserts the list of patterns from the given file into the Trie.
192  // The pattern list file should contain one pattern per line in UTF-8 format.
193  //
194  // Each pattern can contain any non-whitespace characters, however only the
195  // patterns that contain characters from the unicharset of the corresponding
196  // language will be useful.
197  // The only meta character is '\'. To be used in a pattern as an ordinary
198  // string it should be escaped with '\' (e.g. string "C:\Documents" should
199  // be written in the patterns file as "C:\\Documents").
200  // This function supports a very limited regular expression syntax. One can
201  // express a character, a certain character class and a number of times the
202  // entity should be repeated in the pattern.
203  //
204  // To denote a character class use one of:
205  // \c - unichar for which UNICHARSET::get_isalpha() is true (character)
206  // \d - unichar for which UNICHARSET::get_isdigit() is true
207  // \n - unichar for which UNICHARSET::get_isdigit() and
208  // UNICHARSET::isalpha() are true
209  // \p - unichar for which UNICHARSET::get_ispunct() is true
210  // \a - unichar for which UNICHARSET::get_islower() is true
211  // \A - unichar for which UNICHARSET::get_isupper() is true
212  //
213  // \* could be specified after each character or pattern to indicate that
214  // the character/pattern can be repeated any number of times before the next
215  // character/pattern occurs.
216  //
217  // Examples:
218  // 1-8\d\d-GOOG-411 will be expanded to strings:
219  // 1-800-GOOG-411, 1-801-GOOG-411, ... 1-899-GOOG-411.
220  //
221  // http://www.\n\*.com will be expanded to strings like:
222  // http://www.a.com http://www.a123.com ... http://www.ABCDefgHIJKLMNop.com
223  //
224  // Note: In choosing which patterns to include please be aware of the fact
225  // providing very generic patterns will make tesseract run slower.
226  // For example \n\* at the beginning of the pattern will make Tesseract
227  // consider all the combinations of proposed character choices for each
228  // of the segmentations, which will be unacceptably slow.
229  // Because of potential problems with speed that could be difficult to
230  // identify, each user pattern has to have at least kSaneNumConcreteChars
231  // concrete characters from the unicharset at the beginning.
232  bool read_pattern_list(const char *filename, const UNICHARSET &unicharset);

274  // The structure of an EDGE_REF for Trie edges is as follows:
275  // [LETTER_START_BIT, flag_start_bit_):
276  // edge index in *_edges in a TRIE_NODE_RECORD
277  // [flag_start_bit, 30th bit]: node index in nodes (TRIE_NODES vector)
278  //
279  // With this arrangement there are enough bits to represent edge indices
280  // (each node can have at most unicharset_size_ forward edges and
281  // the position of flag_start_bit is set to be log2(unicharset_size_)).
282  // It is also possible to accommodate a maximum number of nodes that is at
283  // least as large as that of the SquishedDawg representation (in SquishedDawg
284  // each EDGE_RECORD has 32-(flag_start_bit+NUM_FLAG_BITS) bits to represent
285  // the next node index).
286  //
287 
288  // Returns the pointer to EDGE_RECORD after decoding the location
289  // of the edge from the information in the given EDGE_REF.
290  // This function assumes that EDGE_REF holds valid node/edge indices.
291  inline EDGE_RECORD *deref_edge_ref(EDGE_REF edge_ref) const {
292  int edge_index = static_cast<int>(
293  (edge_ref & letter_mask_) >> LETTER_START_BIT);
294  int node_index = static_cast<int>(
295  (edge_ref & deref_node_index_mask_) >> flag_start_bit_);
296  TRIE_NODE_RECORD *node_rec = nodes_[node_index];
297  return &(node_rec->forward_edges[edge_index]);
298  }
300  inline EDGE_REF make_edge_ref(NODE_REF node_index,
301  EDGE_INDEX edge_index) const {
302  return ((node_index << flag_start_bit_) |
303  (edge_index << LETTER_START_BIT));
304  }
306  inline void link_edge(EDGE_RECORD *edge, NODE_REF nxt, bool repeats,
307  int direction, bool word_end, UNICHAR_ID unichar_id) {
308  EDGE_RECORD flags = 0;
309  if (repeats) flags |= MARKER_FLAG;
310  if (word_end) flags |= WERD_END_FLAG;
311  if (direction == BACKWARD_EDGE) flags |= DIRECTION_FLAG;
312  *edge = ((nxt << next_node_start_bit_) |
313  (static_cast<EDGE_RECORD>(flags) << flag_start_bit_) |
314  (static_cast<EDGE_RECORD>(unichar_id) << LETTER_START_BIT));
315  }
317  inline void print_edge_rec(const EDGE_RECORD &edge_rec) const {
318  tprintf("|" REFFORMAT "|%s%s%s|%d|", next_node_from_edge_rec(edge_rec),
319  marker_flag_from_edge_rec(edge_rec) ? "R," : "",
320  (direction_from_edge_rec(edge_rec) == FORWARD_EDGE) ? "F" : "B",
321  end_of_word_from_edge_rec(edge_rec) ? ",E" : "",
322  unichar_id_from_edge_rec(edge_rec));
323  }
