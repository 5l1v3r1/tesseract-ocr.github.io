233 int account_splits(const SEAM *seam, const TWERD *word, int blob_index,
234  int blob_direction) {
235  inT8 found_em[3];
236  inT8 width;
237 
238  found_em[0] = seam->split1 == NULL;
239  found_em[1] = seam->split2 == NULL;
240  found_em[2] = seam->split3 == NULL;
241  if (found_em[0] && found_em[1] && found_em[2])
242  return 0;
243  width = 0;
244  do {
245  TBLOB* blob = word->blobs[blob_index];
246  if (!found_em[0])
247  found_em[0] = find_split_in_blob(seam->split1, blob);
248  if (!found_em[1])
249  found_em[1] = find_split_in_blob(seam->split2, blob);
250  if (!found_em[2])
251  found_em[2] = find_split_in_blob(seam->split3, blob);
252  if (found_em[0] && found_em[1] && found_em[2]) {
253  return width;
254  }
255  width++;
256  blob_index += blob_direction;
257  } while (0 <= blob_index && blob_index < word->NumBlobs());
258  return -1;
259 }
