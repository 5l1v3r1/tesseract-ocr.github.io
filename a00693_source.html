128 bool PageIterator::Next(PageIteratorLevel level) {
129  if (it_->block() == NULL) return false; // Already at the end!
130  if (it_->word() == NULL)
131  level = RIL_BLOCK;
132 
133  switch (level) {
134  case RIL_BLOCK:
135  it_->forward_block();
136  break;
137  case RIL_PARA:
138  it_->forward_paragraph();
139  break;
140  case RIL_TEXTLINE:
141  for (it_->forward_with_empties(); it_->row() == it_->prev_row();
142  it_->forward_with_empties());
143  break;
144  case RIL_WORD:
145  it_->forward_with_empties();
146  break;
147  case RIL_SYMBOL:
148  if (cblob_it_ != NULL)
149  cblob_it_->forward();
150  ++blob_index_;
151  if (blob_index_ >= word_length_)
152  it_->forward_with_empties();
153  else
154  return true;
155  break;
156  }
157  BeginWord(0);
158  return it_->block() != NULL;
159 }

166 bool PageIterator::IsAtBeginningOf(PageIteratorLevel level) const {
167  if (it_->block() == NULL) return false; // Already at the end!
168  if (it_->word() == NULL) return true; // In an image block.
169  switch (level) {
170  case RIL_BLOCK:
171  return blob_index_ == 0 && it_->block() != it_->prev_block();
172  case RIL_PARA:
173  return blob_index_ == 0 &&
174  (it_->block() != it_->prev_block() ||
175  it_->row()->row->para() != it_->prev_row()->row->para());
176  case RIL_TEXTLINE:
177  return blob_index_ == 0 && it_->row() != it_->prev_row();
178  case RIL_WORD:
179  return blob_index_ == 0;
180  case RIL_SYMBOL:
181  return true;
182  }
183  return false;
184 }

190 bool PageIterator::IsAtFinalElement(PageIteratorLevel level,
191  PageIteratorLevel element) const {
192  if (Empty(element)) return true; // Already at the end!
193  // The result is true if we step forward by element and find we are
194  // at the the end of the page or at beginning of *all* levels in:
195  // [level, element).
196  // When there is more than one level difference between element and level,
197  // we could for instance move forward one symbol and still be at the first
198  // word on a line, so we also have to be at the first symbol in a word.
199  PageIterator next(*this);
200  next.Next(element);
201  if (next.Empty(element)) return true; // Reached the end of the page.
202  while (element > level) {
203  element = static_cast<PageIteratorLevel>(element - 1);
204  if (!next.IsAtBeginningOf(element))
205  return false;
206  }
207  return true;
208 }

228 // Coordinate system:
229 // Integer coordinates are at the cracks between the pixels.
230 // The top-left corner of the top-left pixel in the image is at (0,0).

246 bool PageIterator::BoundingBoxInternal(PageIteratorLevel level,
247  int* left, int* top,
248  int* right, int* bottom) const {
249  if (Empty(level))
250  return false;
251  TBOX box;
252  PARA *para = NULL;
253  switch (level) {
254  case RIL_BLOCK:
255  box = it_->block()->block->bounding_box();
256  break;
257  case RIL_PARA:
258  para = it_->row()->row->para();
259  // explicit fall-through.
260  case RIL_TEXTLINE:
261  box = it_->row()->row->bounding_box();
262  break;
263  case RIL_WORD:
264  box = it_->word()->word->bounding_box();
265  break;
266  case RIL_SYMBOL:
267  if (cblob_it_ == NULL)
268  box = it_->word()->box_word->BlobBox(blob_index_);
269  else
270  box = cblob_it_->data()->bounding_box();
271  }
272  if (level == RIL_PARA) {
273  PageIterator other = *this;
274  other.Begin();
275  do {
276  if (other.it_->block() &&
277  other.it_->block()->block == it_->block()->block &&
278  other.it_->row() && other.it_->row()->row &&
279  other.it_->row()->row->para() == para) {
280  box = box.bounding_union(other.it_->row()->row->bounding_box());
281  }
282  } while (other.Next(RIL_TEXTLINE));
283  }
284  if (level != RIL_SYMBOL || cblob_it_ != NULL)
285  box.rotate(it_->block()->block->re_rotation());
286  // Now we have a box in tesseract coordinates relative to the image rectangle,
287  // we have to convert the coords to a top-down system.
288  const int pix_height = pixGetHeight(tesseract_->pix_binary());
289  const int pix_width = pixGetWidth(tesseract_->pix_binary());
290  *left = ClipToRange(static_cast<int>(box.left()), 0, pix_width);
291  *top = ClipToRange(pix_height - box.top(), 0, pix_height);
292  *right = ClipToRange(static_cast<int>(box.right()), *left, pix_width);
293  *bottom = ClipToRange(pix_height - box.bottom(), *top, pix_height);
294  return true;
295 }

309 bool PageIterator::BoundingBox(PageIteratorLevel level, const int padding,
310  int* left, int* top,
311  int* right, int* bottom) const {
312  if (!BoundingBoxInternal(level, left, top, right, bottom))
313  return false;
314  // Convert to the coordinate system of the original image.
315  *left = ClipToRange(*left / scale_ + rect_left_ - padding,
316  rect_left_, rect_left_ + rect_width_);
317  *top = ClipToRange(*top / scale_ + rect_top_ - padding,
318  rect_top_, rect_top_ + rect_height_);
319  *right = ClipToRange((*right + scale_ - 1) / scale_ + rect_left_ + padding,
320  *left, rect_left_ + rect_width_);
321  *bottom = ClipToRange((*bottom + scale_ - 1) / scale_ + rect_top_ + padding,
322  *top, rect_top_ + rect_height_);
323  return true;
324 }

386 Pix* PageIterator::GetBinaryImage(PageIteratorLevel level) const {
387  int left, top, right, bottom;
388  if (!BoundingBoxInternal(level, &left, &top, &right, &bottom))
389  return NULL;
390  Pix* pix = NULL;
391  switch (level) {
392  case RIL_BLOCK:
393  case RIL_PARA:
394  int bleft, btop, bright, bbottom;
395  BoundingBoxInternal(RIL_BLOCK, &bleft, &btop, &bright, &bbottom);
396  pix = it_->block()->block->render_mask();
397  // AND the mask and the image.
398  pixRasterop(pix, 0, 0, pixGetWidth(pix), pixGetHeight(pix),
399  PIX_SRC & PIX_DST, tesseract_->pix_binary(),
400  bleft, btop);
401  if (level == RIL_PARA) {
402  // RIL_PARA needs further attention:
403  // clip the paragraph from the block mask.
404  Box* box = boxCreate(left - bleft, top - btop,
405  right - left, bottom - top);
406  Pix* pix2 = pixClipRectangle(pix, box, NULL);
407  boxDestroy(&box);
408  pixDestroy(&pix);
409  pix = pix2;
410  }
411  break;
412  case RIL_TEXTLINE:
413  case RIL_WORD:
414  case RIL_SYMBOL:
415  if (level == RIL_SYMBOL && cblob_it_ != NULL &&
416  cblob_it_->data()->area() != 0)
417  return cblob_it_->data()->render();
418  // Just clip from the bounding box.
419  Box* box = boxCreate(left, top, right - left, bottom - top);
420  pix = pixClipRectangle(tesseract_->pix_binary(), box, NULL);
421  boxDestroy(&box);
422  break;
423  }
424  return pix;
425 }

438 Pix* PageIterator::GetImage(PageIteratorLevel level, int padding,
439  Pix* original_img,
440  int* left, int* top) const {
441  int right, bottom;
442  if (!BoundingBox(level, left, top, &right, &bottom))
443  return NULL;
444  if (original_img == NULL)
445  return GetBinaryImage(level);
446 
447  // Expand the box.
448  *left = MAX(*left - padding, 0);
449  *top = MAX(*top - padding, 0);
450  right = MIN(right + padding, rect_width_);
451  bottom = MIN(bottom + padding, rect_height_);
452  Box* box = boxCreate(*left, *top, right - *left, bottom - *top);
453  Pix* grey_pix = pixClipRectangle(original_img, box, NULL);
454  boxDestroy(&box);
455  if (level == RIL_BLOCK) {
456  Pix* mask = it_->block()->block->render_mask();
457  Pix* expanded_mask = pixCreate(right - *left, bottom - *top, 1);
458  pixRasterop(expanded_mask, padding, padding,
459  pixGetWidth(mask), pixGetHeight(mask),
460  PIX_SRC, mask, 0, 0);
461  pixDestroy(&mask);
462  pixDilateBrick(expanded_mask, expanded_mask, 2*padding + 1, 2*padding + 1);
463  pixInvert(expanded_mask, expanded_mask);
464  pixSetMasked(grey_pix, expanded_mask, MAX_UINT32);
465  pixDestroy(&expanded_mask);
466  }
467  return grey_pix;
468 }

475 bool PageIterator::Baseline(PageIteratorLevel level,
476  int* x1, int* y1, int* x2, int* y2) const {
477  if (it_->word() == NULL) return false; // Already at the end!
478  ROW* row = it_->row()->row;
479  WERD* word = it_->word()->word;
480  TBOX box = (level == RIL_WORD || level == RIL_SYMBOL)
481  ? word->bounding_box()
482  : row->bounding_box();
483  int left = box.left();
484  ICOORD startpt(left, static_cast<inT16>(row->base_line(left) + 0.5));
485  int right = box.right();
486  ICOORD endpt(right, static_cast<inT16>(row->base_line(right) + 0.5));
487  // Rotate to image coordinates and convert to global image coords.
488  startpt.rotate(it_->block()->block->re_rotation());
489  endpt.rotate(it_->block()->block->re_rotation());
490  *x1 = startpt.x() / scale_ + rect_left_;
491  *y1 = (rect_height_ - startpt.y()) / scale_ + rect_top_;
492  *x2 = endpt.x() / scale_ + rect_left_;
493  *y2 = (rect_height_ - endpt.y()) / scale_ + rect_top_;
494  return true;
495 }

563 void PageIterator::BeginWord(int offset) {
564  WERD_RES* word_res = it_->word();
565  if (word_res == NULL) {
566  // This is a non-text block, so there is no word.
567  word_length_ = 0;
568  blob_index_ = 0;
569  word_ = NULL;
570  return;
571  }
572  if (word_res->best_choice != NULL) {
573  // Recognition has been done, so we are using the box_word, which
574  // is already baseline denormalized.
575  word_length_ = word_res->best_choice->length();
576  if (word_res->box_word != NULL) {
577  if (word_res->box_word->length() != word_length_) {
578  tprintf("Corrupted word! best_choice[len=%d] = %s, box_word[len=%d]: ",
579  word_length_, word_res->best_choice->unichar_string().string(),
580  word_res->box_word->length());
581  word_res->box_word->bounding_box().print();
582  }
583  ASSERT_HOST(word_res->box_word->length() == word_length_);
584  }
585  word_ = NULL;
586  // We will be iterating the box_word.
587  if (cblob_it_ != NULL) {
588  delete cblob_it_;
589  cblob_it_ = NULL;
590  }
591  } else {
592  // No recognition yet, so a "symbol" is a cblob.
593  word_ = word_res->word;
594  ASSERT_HOST(word_->cblob_list() != NULL);
595  word_length_ = word_->cblob_list()->length();
596  if (cblob_it_ == NULL) cblob_it_ = new C_BLOB_IT;
597  cblob_it_->set_to_list(word_->cblob_list());
598  }
599  for (blob_index_ = 0; blob_index_ < offset; ++blob_index_) {
600  if (cblob_it_ != NULL)
601  cblob_it_->forward();
602  }
603 }
