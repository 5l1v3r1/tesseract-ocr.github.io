28 // Fixxht overview.
29 // Premise: Initial estimate of x-height is adequate most of the time, but
30 // occasionally it is incorrect. Most notable causes of failure are:
31 // 1. Small caps, where the top of the caps is the same as the body text
32 // xheight. For small caps words the xheight needs to be reduced to correctly
33 // recognize the caps in the small caps word.
34 // 2. All xheight lines, such as summer. Here the initial estimate will have
35 // guessed that the blob tops are caps and will have placed the xheight too low.
36 // 3. Noise/logos beside words, or changes in font size on a line. Such
37 // things can blow the statistics and cause an incorrect estimate.
38 //
39 // Algorithm.
40 // Compare the vertical position (top only) of alphnumerics in a word with
41 // the range of positions in training data (in the unicharset).
42 // See CountMisfitTops. If any characters disagree sufficiently with the
43 // initial xheight estimate, then recalculate the xheight, re-run OCR on
44 // the word, and if the number of vertical misfits goes down, along with
45 // either the word rating or certainty, then keep the new xheight.
46 // The new xheight is calculated as follows:ComputeCompatibleXHeight
47 // For each alphanumeric character that has a vertically misplaced top
48 // (a misfit), yet its bottom is within the acceptable range (ie it is not
49 // likely a sub-or super-script) calculate the range of acceptable xheight
50 // positions from its range of tops, and give each value in the range a
51 // number of votes equal to the distance of its top from its acceptance range.
52 // The x-height position with the median of the votes becomes the new
53 // x-height. This assumes that most characters will be correctly recognized
54 // even if the x-height is incorrect. This is not a terrible assumption, but
55 // it is not great. An improvement would be to use a classifier that does
56 // not care about vertical position or scaling at all.
57 
58 // If the max-min top of a unicharset char is bigger than kMaxCharTopRange
59 // then the char top cannot be used to judge misfits or suggest a new top.
60 const int kMaxCharTopRange = 48;

62 // Returns the number of misfit blob tops in this word.
63 int Tesseract::CountMisfitTops(WERD_RES *word_res) {
64  int bad_blobs = 0;
65  int num_blobs = word_res->rebuild_word->NumBlobs();
66  for (int blob_id = 0; blob_id < num_blobs; ++blob_id) {
67  TBLOB* blob = word_res->rebuild_word->blobs[blob_id];
68  UNICHAR_ID class_id = word_res->best_choice->unichar_id(blob_id);
69  if (unicharset.get_isalpha(class_id) || unicharset.get_isdigit(class_id)) {
70  int top = blob->bounding_box().top();
71  if (top >= INT_FEAT_RANGE)
72  top = INT_FEAT_RANGE - 1;
73  int min_bottom, max_bottom, min_top, max_top;
74  unicharset.get_top_bottom(class_id, &min_bottom, &max_bottom,
75  &min_top, &max_top);
76  if (max_top - min_top > kMaxCharTopRange)
77  continue;
78  bool bad = top < min_top - x_ht_acceptance_tolerance ||
79  top > max_top + x_ht_acceptance_tolerance;
80  if (bad)
81  ++bad_blobs;
82  if (debug_x_ht_level >= 1) {
83  tprintf("Class %s is %s with top %d vs limits of %d->%d, +/-%d\n",
84  unicharset.id_to_unichar(class_id),
85  bad ? "Misfit" : "OK", top, min_top, max_top,
86  static_cast<int>(x_ht_acceptance_tolerance));
87  }
88  }
89  }
90  return bad_blobs;
91 }

93 // Returns a new x-height maximally compatible with the result in word_res.
94 // See comment above for overall algorithm.
95 float Tesseract::ComputeCompatibleXheight(WERD_RES *word_res) {
96  STATS top_stats(0, MAX_UINT8);
97  int num_blobs = word_res->rebuild_word->NumBlobs();
98  for (int blob_id = 0; blob_id < num_blobs; ++blob_id) {
99  TBLOB* blob = word_res->rebuild_word->blobs[blob_id];
100  UNICHAR_ID class_id = word_res->best_choice->unichar_id(blob_id);
101  if (unicharset.get_isalpha(class_id) || unicharset.get_isdigit(class_id)) {
102  int top = blob->bounding_box().top();
103  // Clip the top to the limit of normalized feature space.
104  if (top >= INT_FEAT_RANGE)
105  top = INT_FEAT_RANGE - 1;
106  int bottom = blob->bounding_box().bottom();
107  int min_bottom, max_bottom, min_top, max_top;
108  unicharset.get_top_bottom(class_id, &min_bottom, &max_bottom,
109  &min_top, &max_top);
110  // Chars with a wild top range would mess up the result so ignore them.
111  if (max_top - min_top > kMaxCharTopRange)
112  continue;
113  int misfit_dist = MAX((min_top - x_ht_acceptance_tolerance) - top,
114  top - (max_top + x_ht_acceptance_tolerance));
115  int height = top - kBlnBaselineOffset;
116  if (debug_x_ht_level >= 20) {
117  tprintf("Class %s: height=%d, bottom=%d,%d top=%d,%d, actual=%d,%d : ",
118  unicharset.id_to_unichar(class_id),
119  height, min_bottom, max_bottom, min_top, max_top,
120  bottom, top);
121  }
122  // Use only chars that fit in the expected bottom range, and where
123  // the range of tops is sensibly near the xheight.
124  if (min_bottom <= bottom + x_ht_acceptance_tolerance &&
125  bottom - x_ht_acceptance_tolerance <= max_bottom &&
126  min_top > kBlnBaselineOffset &&
127  max_top - kBlnBaselineOffset >= kBlnXHeight &&
128  misfit_dist > 0) {
129  // Compute the x-height position using proportionality between the
130  // actual height and expected height.
131  int min_xht = DivRounded(height * kBlnXHeight,
132  max_top - kBlnBaselineOffset);
133  int max_xht = DivRounded(height * kBlnXHeight,
134  min_top - kBlnBaselineOffset);
135  if (debug_x_ht_level >= 20) {
136  tprintf(" xht range min=%d, max=%d\n",
137  min_xht, max_xht);
138  }
139  // The range of expected heights gets a vote equal to the distance
140  // of the actual top from the expected top.
141  for (int y = min_xht; y <= max_xht; ++y)
142  top_stats.add(y, misfit_dist);
143  } else if (debug_x_ht_level >= 20) {
144  tprintf(" already OK\n");
145  }
146  }
147  }
148  if (top_stats.get_total() == 0)
149  return 0.0f;
150  // The new xheight is just the median vote, which is then scaled out
151  // of BLN space back to pixel space to get the x-height in pixel space.
152  float new_xht = top_stats.median();
153  if (debug_x_ht_level >= 20) {
154  tprintf("Median xht=%f\n", new_xht);
155  tprintf("Mode20:A: New x-height = %f (norm), %f (orig)\n",
156  new_xht, new_xht / word_res->denorm.y_scale());
157  }
158  // The xheight must change by at least x_ht_min_change to be used.
159  if (fabs(new_xht - kBlnXHeight) >= x_ht_min_change)
160  return new_xht / word_res->denorm.y_scale();
161  else
162  return 0.0f;
163 }
