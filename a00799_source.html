57 /**********************************************************************
58  * make_edgept
59  *
60  * Create an EDGEPT and hook it into an existing list of edge points.
61  **********************************************************************/
62 EDGEPT *make_edgept(int x, int y, EDGEPT *next, EDGEPT *prev) {
63  EDGEPT *this_edgept;
64  /* Create point */
65  this_edgept = new EDGEPT;
66  this_edgept->pos.x = x;
67  this_edgept->pos.y = y;
68  // Now deal with the src_outline steps.
69  C_OUTLINE* prev_ol = prev->src_outline;
70  if (prev_ol != NULL && prev->next == next) {
71  // Compute the fraction of the segment that is being cut.
72  FCOORD segment_vec(next->pos.x - prev->pos.x, next->pos.y - prev->pos.y);
73  FCOORD target_vec(x - prev->pos.x, y - prev->pos.y);
74  double cut_fraction = target_vec.length() / segment_vec.length();
75  // Get the start and end at the step level.
76  ICOORD step_start = prev_ol->position_at_index(prev->start_step);
77  int end_step = prev->start_step + prev->step_count;
78  int step_length = prev_ol->pathlength();
79  ICOORD step_end = prev_ol->position_at_index(end_step % step_length);
80  ICOORD step_vec = step_end - step_start;
81  double target_length = step_vec.length() * cut_fraction;
82  // Find the point on the segment that gives the length nearest to target.
83  int best_step = prev->start_step;
84  ICOORD total_step(0, 0);
85  double best_dist = target_length;
86  for (int s = prev->start_step; s < end_step; ++s) {
87  total_step += prev_ol->step(s % step_length);
88  double dist = fabs(target_length - total_step.length());
89  if (dist < best_dist) {
90  best_dist = dist;
91  best_step = s + 1;
92  }
93  }
94  // The new point is an intermediate point.
95  this_edgept->src_outline = prev_ol;
96  this_edgept->step_count = end_step - best_step;
97  this_edgept->start_step = best_step % step_length;
98  prev->step_count = best_step - prev->start_step;
99  } else {
100  // The new point is poly only.
101  this_edgept->src_outline = NULL;
102  this_edgept->step_count = 0;
103  this_edgept->start_step = 0;
104  }
105  /* Hook it up */
106  this_edgept->next = next;
107  this_edgept->prev = prev;
108  prev->next = this_edgept;
109  next->prev = this_edgept;
110  /* Set up vec entries */
111  this_edgept->vec.x = this_edgept->next->pos.x - x;
112  this_edgept->vec.y = this_edgept->next->pos.y - y;
113  this_edgept->prev->vec.x = x - this_edgept->prev->pos.x;
114  this_edgept->prev->vec.y = y - this_edgept->prev->pos.y;
115  return this_edgept;
116 }

118 /**********************************************************************
119  * remove_edgept
120  *
121  * Remove a given EDGEPT from its list and delete it.
122  **********************************************************************/
123 void remove_edgept(EDGEPT *point) {
124  EDGEPT *prev = point->prev;
125  EDGEPT *next = point->next;
126  // Add point's steps onto prev's steps if they are from the same outline.
127  if (prev->src_outline == point->src_outline && prev->src_outline != NULL) {
128  prev->step_count += point->step_count;
129  }
130  prev->next = next;
131  next->prev = prev;
132  prev->vec.x = next->pos.x - prev->pos.x;
133  prev->vec.y = next->pos.y - prev->pos.y;
134  delete point;
135 }

167  * split_outline
168  *
169  * Split between these two edge points.
170  **********************************************************************/
171 void split_outline(EDGEPT *join_point1, EDGEPT *join_point2) {
172  assert(join_point1 != join_point2);
173 
174  EDGEPT* temp2 = join_point2->next;
175  EDGEPT* temp1 = join_point1->next;
176  /* Create two new points */
177  EDGEPT* new_point1 = make_edgept(join_point1->pos.x, join_point1->pos.y,
178  temp1, join_point2);
179  EDGEPT* new_point2 = make_edgept(join_point2->pos.x, join_point2->pos.y,
180  temp2, join_point1);
181  // Join_point1 and 2 are now cross-over points, so they must have NULL
182  // src_outlines and give their src_outline information their new
183  // replacements.
184  new_point1->src_outline = join_point1->src_outline;
185  new_point1->start_step = join_point1->start_step;
186  new_point1->step_count = join_point1->step_count;
187  new_point2->src_outline = join_point2->src_outline;
188  new_point2->start_step = join_point2->start_step;
189  new_point2->step_count = join_point2->step_count;
190  join_point1->src_outline = NULL;
191  join_point1->start_step = 0;
192  join_point1->step_count = 0;
193  join_point2->src_outline = NULL;
194  join_point2->start_step = 0;
195  join_point2->step_count = 0;
196  join_point1->MarkChop();
197  join_point2->MarkChop();
198 }
